# new

function _new (Fnc) {
  return function () {
    // const obj = {
    //   __proto__: Fnc.prototype
    // }
    const obj = {};
    Object.defineProperty(obj, '__proto__', {
      enumerable: false,
      value: Fnc.prototype
    })
    Fnc.apply(obj, arguments);
    return obj;
  }
}
